 11 	/**
 12 	 * Helper function to traverse an array using for loop
 13 	 * @param aArray
 14 	 * @param fpProcess
 15 	 */
 16 	function loop(aArray, fpProcess)
 17 	{
 18 		var nItem = 0;
 19 		var nLenItem = aArray.length;
 20 		var oItem = null;
 21 		for(nItem = 0; nItem < nLenItem; nItem = nItem + 1)
 22 		{
 23 			oItem = aArray[nItem];
 24 			fpProcess(oItem);
 25 		}
 26 	}

 28 	/**
 29 	 * Class to create interfaces to be implemented in other classes
 30 	 * @class Represents an Interface
 31 	 * @constructor
 32 	 * @name Interface
 33 	 * @author Tomas Corral Casas
 34 	 * @version 1.0
 35 	 * @param aMethods
 36 	 */
 37 	var Interface = function (aMethods)
 38 	{
 39 		if(typeof aMethods === 'string')
 40 		{
 41 			aMethods = [aMethods];
 42 		}
 43 		this.aMethods = aMethods || [];
 44 	};
 45 	/**
 46 	 * Implement is a static method to implement the oInterface methods in oTarget
 47 	 * Default behaviour when implement an interface is to throw an error if the class
 48 	 * doesn't implement all the methods to be implemented
 49 	 * If bAllowAbstract is set this implementation doesn't throw an error when one or
 50 	 * some methods are not implemented, but it will return an abstract class.
 51 	 * @static
 52 	 * @member Interface
 53 	 * @param oInterface
 54 	 * @param oTarget
 55 	 * @param bAllowAbstract
 56 	 * @return {Function} Wrapped oTarget with the interface implemented
 57 	 */
 58 	Interface.implement = function(oInterface, oTarget, bAllowAbstract)
 59 	{
 60 		return oInterface.implementIn(oTarget, bAllowAbstract);
 61 	};
 62 	/**
 63 	 * implementIn is the method to be used when implementing an interface in one class (oTarget)
 64 	 * When the interface is implemented the method creates a callback that will throw an error if the method is not overwritten.
 65 	 * Also adds a check to the constructor of the oTarget to know if all the methods are implemented.
 66 	 * This check by default throws an error if one of the interface methods is not implemented on the class which implements the interface.
 67 	 * If you want to implement an interface but you want to return an abstract class you will need to pass a boolean (true).
 68 	 * @member Interface.prototype
 69 	 * @param oTarget
 70 	 * @param bAllowAbstract
 71 	 * @return {Function} Wrapped oTarget with the interface implemented
 72 	 */
 73 	Interface.prototype.implementIn = function(oTarget, bAllowAbstract)
 74 	{
 75 		var aMethods = this.aMethods;
 76 		var fpInterfaceMethod = function()
 77 		{
 78 			throw new Error("This method must be overwritten !");
 79 		};
 80 		var fpMethod = null;
 81 		loop(aMethods, function(sMethod)
 82 		{
 83 			fpMethod = oTarget.prototype[sMethod];
 84 			if(!fpMethod)
 85 			{
 86 				oTarget.prototype[sMethod] = fpInterfaceMethod;
 87 			}
 88 		});
 89 		if(!oTarget.__aImplementedMethods__)
 90 		{
 91 			oTarget.__aImplementedMethods__ = [];
 92 		}
 93 		[].push.apply(oTarget.__aImplementedMethods__, aMethods);
 94 		var oInterface = function()
 95 		{
 96 			loop(oTarget.__aImplementedMethods__, function(sMethod)
 97 			{
 98 				fpMethod = oInterface.prototype[sMethod];
 99 				if(fpMethod === fpInterfaceMethod && !bAllowAbstract)
100 				{
101 					throw new Error("This class implemented some interface but doesn't implemented at least one of them. Method not implemented: " + sMethod);
102 				}
103 			});
104 			oTarget.apply(this, arguments);
105 		};
106 		oInterface.prototype = oTarget.prototype;
107 		oInterface.constructor = oTarget;
108 		return oInterface;
109 	};
110 	/*
111 	 * Expose Interface Class to global
112 	 */
113 	global.Interface = Interface;
